var h_header = $('header').height();
	var isMobile = document.documentElement.clientWidth;
    if (isMobile < 768) {
        $('#fullpage').fullpage ({
            resize: false,
            responsiveWidth: 1000,
            fixedElements: 'header, .panel',
            touchSensitivity: 20,
			responsiveHeight: 768,
			scrollOverflow: false,
			navigation: true,
			pauseOnHover: true,
			slidesNavigation: true,
			controlArrows: false,
            afterRender: function () {
	            interval = setInterval(function() {
		            $.fn.fullpage.moveSlideRight();
		        }, 5000);
	        },
	       
        });     
    }
    else {
		$('#fullpage').fullpage({
			navigation: true,
			controlArrows: false,
			fixedElements: 'header, .panel',
			scrollOverflow: true,
			paddingTop: h_header,
			touchSensitivity: 20,
			responsiveWidth: 1000,
			slidesNavigation: true,
			pauseOnHover: true,
			responsiveHeight: 768,
			
	        afterRender: function () {
	            interval = setInterval(function() {
		            $.fn.fullpage.moveSlideRight();
		        }, 4000);
	        },
	        
		});
	}
			
	$('#fp-nav').append('<button class="section-next">next</button>');
	$('.section-next').click(function(){
	    $.fn.fullpage.moveSectionDown();
	});


	$('#fullpage .slide .wrapper').mouseover(function() {
		clearInterval(interval);
		interval = null;
	})
	$('#fullpage .slide .wrapper').mouseout(function() {
		 interval = setInterval(function() {
            $.fn.fullpage.moveSlideRight();
        }, 4000);
	});
