 var vid = $("#bgvid");
	var pauseButton = $(".stop-play");

	vid.get(0).play();
	pauseButton.on("click", function() {
		if ($("#bgvid").length) {
			// $(this).toggleClass('');
			if (!(vid.get(0).paused)) {
				vid.get(0).pause();
				$(this).html('<i class="zmdi zmdi-play"></i> PLAY')
			} else {
				vid.get(0).play();
				$(this).html('<i class="zmdi zmdi-pause"></i> PAUSE')

			}
		}
	});
