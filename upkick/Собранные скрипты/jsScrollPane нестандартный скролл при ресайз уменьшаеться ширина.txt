$('.jScrollPane').each(function(){
		$(this).jScrollPane(
			{
				horizontalDragMaxWidth: '47',

			}
		);
		var api = $(this).data('jsp');
		var throttleTimeout;
		$(window).bind('resize',function(){
			if (!throttleTimeout) {
				throttleTimeout = setTimeout(
					function()
					{
						api.reinitialise();
						throttleTimeout = null;
					},
					50
				);
			}
		});
	});
